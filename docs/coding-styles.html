// BFS shortest path
int shortest(const vector<vector<int>>& g,int s,int t){
 queue<int> q;vector<int> d(g.size(),-1);
 d[s]=0;q.push(s);
 while(!q.empty()){
   int u=q.front();q.pop();
   if(u==t)return d[u];
   for(int v:g[u])
     if(d[v]==-1){d[v]=d[u]+1;q.push(v);}
 }
 return-1;
}

// Rust macro that generates a getter
macro_rules! make_getter{
 ($n:ident,$f:ident,$t:ty)=>{
  impl $n{
   pub fn $f(&self)->$t{self.$f}
  }
 };
}
struct Config{port:u16}
make_getter!(Config,port,u16);
fn main(){
 let c=Config{port:8080};
 println!("port = {}",c.port());
}
